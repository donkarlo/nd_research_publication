\begin{frame}[allowframebreaks]{Self-Evaluation}
    \begin{algorithm}[H]
		\caption{Self-evaluation}
        \begin{algorithmic}[1]
			\State Discover periods by autocorrelation.
            \State Use \(70\%\) of the discovered periods for training.
            \State Use \(30\%\) of the discovered periods for testing.
            \State Compute the discrepancy value on the test periods.
            \State Check whether the discrepancy value is below the tolerance value.
        \end{algorithmic}
    \end{algorithm}
\end{frame}
