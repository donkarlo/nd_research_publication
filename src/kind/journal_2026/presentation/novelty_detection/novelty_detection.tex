\begin{frame}{Novelty detection}
    Discrepancy
    \begin{itemize}
        \item \textbf{Anomaly}: Short time over the threshold
        \item \textbf{Novelty}: Continuously hi anomaly
        \item \textbf{Outlier}: Statistically away from the observation distribution but yet a valid reading
    \end{itemize}
    
    \textbf{Anomaly} is the Mahalanobis distance between the prediction and the preprocessed, fused observation.
    
    
    \section{Threshold}
        \begin{enumerate}
            \item Breaking an event-specific-knowledge sensory data into periods
            \item Split the periods into train-test
            \item take $\mu + n \times \sigma$ anomaly values as the threshold
        \end{enumerate}
        
        \begin{algorithm}[H]
		\caption{Novelty Detection from Sensor Sequences}
        \begin{algorithmic}[1]
			\Require Preprocessed sensor sequence $X = \{x_1, x_2, \dots, x_T\}$
            \Require Trained predictive model $f_{\theta}$
            \Ensure Novelty threshold $\tau = \{y_1, y_2, \dots, y_T\}$
            \Ensure Novelty scalar value
            
            
            $Y = \{y_1, y_2, \dots, y_T\}$
            \State \Return $Y$
        \end{algorithmic}
        \end{algorithm}


\end{frame}
