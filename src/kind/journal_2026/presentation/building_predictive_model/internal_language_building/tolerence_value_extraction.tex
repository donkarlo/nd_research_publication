\begin{frame}[allowframebreaks]{Tolerance Value Determination}
    \begin{algorithm}[H]
		\caption{Tolerance value determination}
        \begin{algorithmic}[1]
			\State Build the predictive model for the normal scenario.
            \State Determine discrepancy values for the same scenario.
            \State Take \(\mu + 3\sigma\) as the tolerance value.
        \end{algorithmic}
    \end{algorithm}
\end{frame}
